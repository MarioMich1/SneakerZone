<script>
function toggleUploadForm() {
  document.getElementById('uploadForm').classList.toggle('hidden');
}

function subirTenis() {
  const nombre = document.getElementById('nombreTenis').value;
  const precio = document.getElementById('precioTenis').value;
  const categoria = document.getElementById('categoria').value;
  const marca = document.getElementById('marca').value;
  const tallas = document.getElementById('tallasTenis').value.split(',');
  const foto = document.getElementById('fotoTenis').files[0];

  if (!nombre || !precio || !tallas || !foto) {
    alert('Por favor completa todos los campos y selecciona una imagen.');
    return;
  }

  const reader = new FileReader();
  reader.onload = function(e) {
    const imageUrl = e.target.result;

    const card = document.createElement('div');
    card.className = `product-card nike-hover group cursor-pointer`;
    card.setAttribute('data-brand', marca.toLowerCase());

    card.innerHTML = `
      <div class="relative overflow-hidden bg-gray-100 rounded-lg mb-4">
        <div class="product-image aspect-square flex items-center justify-center p-4">
          <img src="${imageUrl}" alt="${nombre}" class="w-full h-full object-cover rounded-lg">
        </div>
      </div>
      <div class="space-y-2">
        <h3 class="font-medium text-black">${nombre}</h3>
        <p class="text-gray-500 text-sm">${marca} - Tenis para ${categoria.charAt(0).toUpperCase() + categoria.slice(1)}</p>
        <p class="font-medium text-black">$${precio}</p>
        <div class="size-selector mt-4">
          <p class="text-sm text-gray-600 mb-2">Talla:</p>
          <div class="grid grid-cols-4 gap-2 mb-4">
            ${tallas.map(t => `<button class="size-btn border border-gray-300 hover:border-black py-2 text-sm rounded transition-colors" data-size="${t.trim()}">${t.trim()}</button>`).join('')}
          </div>
        </div>
        <button class="whatsapp-btn w-full nike-button bg-green-600 text-white py-3 mt-2 hover:bg-green-700" disabled>
          Comprar por WhatsApp
        </button>
      </div>
    `;

    const idMap = {
      hombre: 'men-catalog',
      mujer: 'women-catalog',
      ninos: 'ninos-page'
    };

    const contenedor = document.querySelector(`#${idMap[categoria]} .grid`);
    if (contenedor) {
      contenedor.appendChild(card);
      alert(`Tenis "${nombre}" cargado correctamente en la categoría ${categoria} (${marca}) con precio $${precio} y tallas: ${tallas.join(', ')}`);
    } else {
      alert('No se encontró la sección correspondiente. Verifica que exista el contenedor con id "' + idMap[categoria] + '".');
    }

    toggleUploadForm();
  };

  reader.readAsDataURL(foto);
}
</script>
